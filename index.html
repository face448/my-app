// Game State Management

class GameState {
    constructor() {
        this.loadState();
    }

    loadState() {
        const saved = localStorage.getItem('nextYouState');
        if (saved) {
            const state = JSON.parse(saved);
            this.tier = state.tier || 'basic';
            this.stats = state.stats || this.getDefaultStats();
            this.actionsRemaining = state.actionsRemaining !== undefined ? state.actionsRemaining : 3;
            this.streak = state.streak || 0;
            this.lastActionDate = state.lastActionDate || null;
            this.avatarLevel = state.avatarLevel || 1;
            this.totalActions = state.totalActions || 0;
        } else {
            this.tier = 'basic';
            this.stats = this.getDefaultStats();
            this.actionsRemaining = 3;
            this.streak = 0;
            this.lastActionDate = null;
            this.avatarLevel = 1;
            this.totalActions = 0;
        }
        this.checkDailyReset();
    }

    getDefaultStats() {
        return {
            strength: 0,
            focus: 0,
            confidence: 0,
            discipline: 0,
            money: 0
        };
    }

    checkDailyReset() {
        const today = new Date().toDateString();
        if (this.lastActionDate !== today) {
            // Reset daily actions
            if (this.tier === 'basic') {
                this.actionsRemaining = 3;
            }
            // Update streak
            if (this.lastActionDate) {
                const lastDate = new Date(this.lastActionDate);
                const currentDate = new Date(today);
                const diffDays = Math.floor((currentDate - lastDate) / (1000 * 60 * 60 * 24));
                
                if (diffDays === 1) {
                    this.streak++;
                } else if (diffDays > 1) {
                    this.streak = 0;
                }
            }
        }
    }

    saveState() {
        const state = {
            tier: this.tier,
            stats: this.stats,
            actionsRemaining: this.actionsRemaining,
            streak: this.streak,
            lastActionDate: this.lastActionDate,
            avatarLevel: this.avatarLevel,
            totalActions: this.totalActions
        };
        localStorage.setItem('nextYouState', JSON.stringify(state));
    }

    updateStats(effects) {
        for (const [stat, value] of Object.entries(effects)) {
            if (this.stats[stat] !== undefined) {
                this.stats[stat] += value;
                // Cap stats at 999
                this.stats[stat] = Math.min(this.stats[stat], 999);
            }
        }
        this.saveState();
    }

    performAction() {
        this.totalActions++;
        if (this.tier === 'basic') {
            this.actionsRemaining = Math.max(0, this.actionsRemaining - 1);
        }
        this.lastActionDate = new Date().toDateString();
        
        // Level up every 10 actions
        const newLevel = Math.floor(this.totalActions / 10) + 1;
        if (newLevel > this.avatarLevel) {
            this.avatarLevel = newLevel;
            return { levelUp: true, newLevel: this.avatarLevel };
        }
        
        this.saveState();
        return { levelUp: false };
    }

    upgradeToPro() {
        this.tier = 'pro';
        this.actionsRemaining = 999; // Unlimited
        this.saveState();
    }

    isPro() {
        return this.tier === 'pro';
    }

    canPerformAction() {
        if (this.tier === 'pro') return true;
        return this.actionsRemaining > 0;
    }
}
